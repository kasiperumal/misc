name: Scheduled Merge: develop-1 → develop

on:
  schedule:
    - cron: '0 4 * * *'  # Runs daily at 9:30 AM IST (4:00 UTC)
  workflow_dispatch:      # Allow manual trigger too

jobs:
  create-pr:
    runs-on: ubuntu-latest

    steps:
      - name: Checkout base branch
        uses: actions/checkout@v4
        with:
          ref: develop

      - name: Configure Git
        run: |
          git config user.name "github-actions[bot]"
          git config user.email "github-actions[bot]@users.noreply.github.com"

      - name: Fetch feature branch
        run: |
          git fetch origin develop-1:develop-1

      - name: Check for fast-forward possibility
        id: check_ff
        run: |
          git checkout develop
          if git merge-base --is-ancestor develop develop-1; then
            echo "Can fast-forward merge"
            echo "ff_possible=true" >> $GITHUB_OUTPUT
          else
            echo "Fast-forward not possible"
            echo "ff_possible=false" >> $GITHUB_OUTPUT
          fi

      - name: Create Pull Request if fast-forward is possible
        if: steps.check_ff.outputs.ff_possible == 'true'
        uses: peter-evans/create-pull-request@v5
        with:
          token: ${{ secrets.GITHUB_TOKEN }}
          title: 'Scheduled Merge: develop-1 → develop'
          body: |
            This PR merges `develop-1` into `develop` using a fast-forward strategy.
            - Auto-created via GitHub Actions.
            - Please review and merge if conflict-free.
          base: develop
          head: develop-1
          draft: false
          reviewers: |
            kasiperumal
