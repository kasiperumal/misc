const currentMilestoneIndex = milestoneInfo.findIndex(milestone => milestone.id === app.milestone);
if (currentMilestoneIndex !== -1 && currentMilestoneIndex < milestoneInfo.length - 1) {
    // Move to next milestone in sequence
    const nextMilestone = milestoneInfo[currentMilestoneIndex + 1];
    app.milestone = nextMilestone.id;
    app.lastProgress = Date.now();
    app.milestoneEntryTime = Date.now();
    app.progressHistory.push(app.milestone);
    
    // Check if reached the final milestone (last item in milestoneInfo array)
    if (currentMilestoneIndex + 1 === milestoneInfo.length - 1) {
        workflowStats.completed++;
    }
    return true;
}
